#ターミナルのコマンド一覧
echo test
touch ~/Desktop/runteq.txt *ファイルの作成
cat ~/Desktop/runteq.txt　*ファイルの中身の表
echo *文字を表示することができるコマンド。>（リダイレクト）というコマンドと組み合わせると、echo コマンドで表示した文字をファイルに書き込むことができる
mv ~/Desktop/runteq.txt ~/Desktop/RUNTEQ.txt *mv」コマンドはファイルやディレクトリを移動させるコマンド
rm ~/Desktop/RUNTEQ.txt *「rm」はファイルやディレクトリを削除できるコマンドです
mkdir * rm ~/Desktop/RUNTEQ.txt 「mkdir」はディレクトリを作成するためのコマンドです。
ls *「ls」は、現在作業しているディレクトリやファイルの一覧を表示するためのコマンドです。
pwd *現在作業しているディレクトリのパスを表示するためのコマンド
cd *「cd」は現在作業しているディレクトリを移動するためのコマンド
mrdir *空のディレクトリを削除できるコマンド

. はカレントディレクトリ、 .. は親ディレクトリを表すことができます
「Tab」キーを押すか、Ctrl + I を押すと、ターミナルは入力中のコマンドの候補を表示してくれます
